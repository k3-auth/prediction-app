function updateBuyModal(){

  const bal=isPaid?coins.paid:coins.free;

  const after=bal-cost;

  const type=isPaid?'💰':'🪙';

  document.getElementById('buy-balance-calc').textContent=`${type} ${bal.toLocaleString()} → 購入後: ${after.toLocaleString()}`;

  const err=after<0;

  document.getElementById('buy-error').style.display=err?'block':'none';

  document.getElementById('btn-confirm-buy').disabled=err;

}
